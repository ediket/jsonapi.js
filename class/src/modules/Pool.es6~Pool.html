    Pool

    Example:

    pool.addRemote('foo', '/foo/');

// Fetch Many Resources
pool.fetch('foo')
.then(resources => {
  resources === pool.get('foo');  // true
});

// Fetch One Resource
pool.fetch('foo', 1)
.then(resource => {
  resource === pool.get('foo', 1);  // true
});

// Create Resource
pool.create('foo', {
  attributes: {
    content: 'bar'
  }
})
.then(resource => {
  resource === pool.get('foo', resource.id);  // true
});

// Update Resource
pool.update('foo', 1, {
  attributes: {
    content: 'bar'
  }
})

// Remove Resource
pool.remove('foo', 1)
.then(() => {
  undefined === pool.get('foo', 1);  // true
});
