    RESTful

    wrapper function to make RESTful ajax request.

    Example:

    // Fetch comment 1
RESTful.get('/comments/1')
.then(response => {
  let resource = new Resource(response.data);
})

// Create comment
RESTful.post('/comments', {
  type: 'comment',
  attributes: {
    content: 'hello world'
  }
})
.then(response => {
  let resource = new Resource(response.data);
})

// Update content of comment 2
RESTful.patch('/comments/2', {
  type: 'comment',
  id: 2,
  attributes: {
    content: 'fix'
  }
})

// Remove comment 2
RESTful.delete('/comments/1')
